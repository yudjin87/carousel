482 //------------------------------------------------------------------------------
483 template <typename TItem>
484 int ObservableList<TItem>::removeAll(const TItem &item)
485 {
486  if (m_items.indexOf(item) == -1)
487  return 0;
488 
489  QList<TItem> items;
490  foreach(const TItem &itemInList, m_items) {
491  bool sameItem = itemInList == item;
492  if (sameItem && m_items.contains(itemInList))
493  items.push_back(itemInList);
494  }
495 
496  if (items.empty())
497  return 0;
498 
499  m_items.removeAll(item);
500 
501  Changes<TItem> changes(Remove, items);
502  notifyObservers(changes);
503 
504  return items.size();
505 }
