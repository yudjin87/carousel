200 //------------------------------------------------------------------------------
201 template<typename TValue>
202 TValue TypeObjectsMap<TValue>::unregisterInstance(const QString &key, const QString &tag)
203 {
204  auto predicate = std::bind(&TypeObjectsMap<TValue>::findPredicate, std::placeholders::_1, key, tag);
205  auto result = std::find_if(m_objects.begin(), m_objects.end(), predicate);
206 
207  if (result == m_objects.end())
208  return TValue();
209 
210  InstanceObject<TValue> * foundInstance = *result;
211 
212  m_objects.erase(result);
213 
214  return foundInstance->instance;
215 }
